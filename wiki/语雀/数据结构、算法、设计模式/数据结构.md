# 数据结构

## 1. 数组 (Array)

### 特点

| 特性 | 说明 |
|------|------|
| 内存连续 | 元素在内存中连续存储 |
| 随机访问 | O(1) 时间访问任意元素 |
| 固定大小 | 静态数组大小固定 |
| 插入删除 | O(n) 时间复杂度 |

### Python 实现

```python
# Python列表就是动态数组
arr = [1, 2, 3, 4, 5]

# 访问元素 - O(1)
print(arr[0])  # 1
print(arr[-1])  # 5

# 查找元素 - O(n)
index = arr.index(3)  # 2

# 插入元素 - O(n)
arr.insert(2, 10)  # [1, 2, 10, 3, 4, 5]

# 删除元素 - O(n)
arr.remove(10)  # [1, 2, 3, 4, 5]

# 尾部添加 - O(1)
arr.append(6)  # [1, 2, 3, 4, 5, 6]

# 尾部删除 - O(1)
arr.pop()  # [1, 2, 3, 4, 5]

# 切片 - O(k)
sub = arr[1:3]  # [2, 3]

# 长度 - O(1)
length = len(arr)  # 5
```

### 二维数组

```python
# 创建二维数组
matrix = [[0 for _ in range(3)] for _ in range(3)]

# 访问
matrix[0][0] = 1
matrix[1][2] = 5

# 遍历
for i in range(len(matrix)):
    for j in range(len(matrix[0])):
        print(matrix[i][j], end=' ')
    print()
```

### 常见操作复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 访问 | O(1) |
| 查找 | O(n) |
| 插入(头部) | O(n) |
| 插入(尾部) | O(1) |
| 删除(头部) | O(n) |
| 删除(尾部) | O(1) |

## 2. 队列 (Queue)

### 特点

| 特性 | 说明 |
|------|------|
| FIFO | 先进先出 |
| 入队 | 在队尾添加元素 |
| 出队 | 从队头移除元素 |

### Python 实现

```python
from collections import deque

# 使用deque实现队列
queue = deque()

# 入队 - O(1)
queue.append(1)
queue.append(2)
queue.append(3)

# 出队 - O(1)
item = queue.popleft()  # 1

# 查看队头 - O(1)
front = queue[0]  # 2

# 长度 - O(1)
length = len(queue)  # 2

# 判断是否为空
is_empty = len(queue) == 0
```

### 队列类实现

```python
class Queue:
    def __init__(self):
        self._items = deque()

    def enqueue(self, item):
        """入队"""
        self._items.append(item)

    def dequeue(self):
        """出队"""
        if self.is_empty():
            raise IndexError("Queue is empty")
        return self._items.popleft()

    def front(self):
        """查看队头"""
        if self.is_empty():
            raise IndexError("Queue is empty")
        return self._items[0]

    def is_empty(self):
        """判断是否为空"""
        return len(self._items) == 0

    def size(self):
        """队列大小"""
        return len(self._items)

    def __str__(self):
        return str(list(self._items))

# 使用
q = Queue()
q.enqueue(1)
q.enqueue(2)
q.enqueue(3)
print(q)        # [1, 2, 3]
print(q.dequeue())  # 1
print(q)        # [2, 3]
```

### 优先队列

```python
import heapq

class PriorityQueue:
    def __init__(self):
        self._heap = []

    def push(self, item, priority):
        """入队(带优先级)"""
        heapq.heappush(self._heap, (priority, item))

    def pop(self):
        """出队(最高优先级)"""
        if self.is_empty():
            raise IndexError("PriorityQueue is empty")
        return heapq.heappop(self._heap)[1]

    def is_empty(self):
        return len(self._heap) == 0

    def size(self):
        return len(self._heap)

# 使用
pq = PriorityQueue()
pq.push("low", 3)
pq.push("high", 1)
pq.push("medium", 2)

print(pq.pop())  # high
print(pq.pop())  # medium
print(pq.pop())  # low
```

## 3. 堆栈 (Stack)

### 特点

| 特性 | 说明 |
|------|------|
| LIFO | 后进先出 |
| 入栈 | 在栈顶添加元素 |
| 出栈 | 从栈顶移除元素 |

### Python 实现

```python
# 使用列表实现栈
stack = []

# 入栈 - O(1)
stack.append(1)
stack.append(2)
stack.append(3)

# 出栈 - O(1)
item = stack.pop()  # 3

# 查看栈顶 - O(1)
top = stack[-1]  # 2

# 长度 - O(1)
length = len(stack)  # 2

# 判断是否为空
is_empty = len(stack) == 0
```

### 堆栈类实现

```python
class Stack:
    def __init__(self):
        self._items = []

    def push(self, item):
        """入栈"""
        self._items.append(item)

    def pop(self):
        """出栈"""
        if self.is_empty():
            raise IndexError("Stack is empty")
        return self._items.pop()

    def peek(self):
        """查看栈顶"""
        if self.is_empty():
            raise IndexError("Stack is empty")
        return self._items[-1]

    def is_empty(self):
        """判断是否为空"""
        return len(self._items) == 0

    def size(self):
        """栈大小"""
        return len(self._items)

    def __str__(self):
        return str(self._items)

# 使用
s = Stack()
s.push(1)
s.push(2)
s.push(3)
print(s)        # [1, 2, 3]
print(s.pop())  # 3
print(s)        # [1, 2]
```

### 常见应用

```python
# 1. 括号匹配
def is_valid_parentheses(s: str) -> bool:
    stack = Stack()
    mapping = {')': '(', '}': '{', ']': '['}

    for char in s:
        if char in mapping:
            if stack.is_empty() or stack.pop() != mapping[char]:
                return False
        else:
            stack.push(char)

    return stack.is_empty()

# 2. 表达式求值
def evaluate_postfix(expression: str) -> int:
    stack = Stack()
    operators = {'+', '-', '*', '/'}

    for token in expression.split():
        if token in operators:
            b = stack.pop()
            a = stack.pop()
            if token == '+': stack.push(a + b)
            elif token == '-': stack.push(a - b)
            elif token == '*': stack.push(a * b)
            elif token == '/': stack.push(int(a / b))
        else:
            stack.push(int(token))

    return stack.pop()

# 3. 浏览器历史
class BrowserHistory:
    def __init__(self):
        self._history = Stack()
        self._forward = Stack()
        self._current = None

    def visit(self, url):
        if self._current:
            self._history.push(self._current)
        self._current = url
        self._forward = Stack()

    def back(self):
        if self._history.is_empty():
            return self._current
        self._forward.push(self._current)
        self._current = self._history.pop()
        return self._current

    def forward(self):
        if self._forward.is_empty():
            return self._current
        self._history.push(self._current)
        self._current = self._forward.pop()
        return self._current
```

## 4. 链表 (Linked List)

### 单链表

```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class LinkedList:
    def __init__(self):
        self.head = None
        self.size = 0

    def append(self, val):
        """在末尾添加节点"""
        new_node = ListNode(val)
        if not self.head:
            self.head = new_node
        else:
            current = self.head
            while current.next:
                current = current.next
            current.next = new_node
        self.size += 1

    def prepend(self, val):
        """在头部添加节点"""
        new_node = ListNode(val)
        new_node.next = self.head
        self.head = new_node
        self.size += 1

    def delete(self, val):
        """删除指定值的节点"""
        if not self.head:
            return

        if self.head.val == val:
            self.head = self.head.next
            self.size -= 1
            return

        current = self.head
        while current.next:
            if current.next.val == val:
                current.next = current.next.next
                self.size -= 1
                return
            current = current.next

    def find(self, val):
        """查找节点"""
        current = self.head
        while current:
            if current.val == val:
                return current
            current = current.next
        return None

    def to_list(self):
        """转换为列表"""
        result = []
        current = self.head
        while current:
            result.append(current.val)
            current = current.next
        return result

    def __len__(self):
        return self.size

    def __str__(self):
        return str(self.to_list())

# 使用
ll = LinkedList()
ll.append(1)
ll.append(2)
ll.append(3)
ll.prepend(0)
print(ll)  # [0, 1, 2, 3]

ll.delete(2)
print(ll)  # [0, 1, 3]
```

### 双向链表

```python
class DoublyListNode:
    def __init__(self, val=0, prev=None, next=None):
        self.val = val
        self.prev = prev
        self.next = next

class DoublyLinkedList:
    def __init__(self):
        self.head = None
        self.tail = None
        self.size = 0

    def append(self, val):
        new_node = DoublyListNode(val)
        if not self.head:
            self.head = new_node
            self.tail = new_node
        else:
            new_node.prev = self.tail
            self.tail.next = new_node
            self.tail = new_node
        self.size += 1

    def prepend(self, val):
        new_node = DoublyListNode(val)
        if not self.head:
            self.head = new_node
            self.tail = new_node
        else:
            new_node.next = self.head
            self.head.prev = new_node
            self.head = new_node
        self.size += 1

    def delete(self, node):
        if node.prev:
            node.prev.next = node.next
        else:
            self.head = node.next

        if node.next:
            node.next.prev = node.prev
        else:
            self.tail = node.prev

        self.size -= 1

    def to_list(self):
        result = []
        current = self.head
        while current:
            result.append(current.val)
            current = current.next
        return result

    def __str__(self):
        return str(self.to_list())
```

### 链表 vs 数组

| 特性 | 数组 | 链表 |
|------|------|------|
| 内存 | 连续 | 分散 |
| 访问 | O(1) | O(n) |
| 插入(头部) | O(n) | O(1) |
| 插入(尾部) | O(1) | O(n) |
| 删除(头部) | O(n) | O(1) |
| 删除(尾部) | O(1) | O(n) |
| 空间 | 额外开销小 | 每个节点额外指针 |
